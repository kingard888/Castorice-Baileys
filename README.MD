    // ğŸ”— Connection Events

        const { connection, lastDisconnect } = update;

            console.log('âŒ Koneksi ditutup. Reconnect:', shouldReconnect);
            
            if(shouldReconnect) {
                connectToWhatsApp();

        } else if(connection === 'open') {
            console.log('âœ… Berhasil terhubung ke WhatsApp!');
        }
    });

    // ğŸ“¨ Message Events
    sock.ev.on('messages.upsert', async (m) => {
        const msg = m.messages[0];
        
        if (!msg.key.fromMe && msg.message) {
            console.log('ğŸ“© Pesan baru:', JSON.stringify(m, undefined, 2));
            
            // ğŸ¤– Echo response
            await sock.sendMessage(msg.key.remoteJid, { 
                text: 'Hello! Terima kasih pesanmu ğŸ‘‹' 
            });
        }
